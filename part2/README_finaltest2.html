def linregeress(x, y):
    res = stats.linregress(x, y)

    def fun(x):  # 拟合后的线性函数
        return res.intercept + res.slope * x

    plt.plot(x, y, 'o', label='original data')  # 原数据标点
    plt.plot(x, fun(x), 'r', label='fitted line')  # 拟合直线
    plt.legend()
    plt.show()

    def func(x, a, c):
        return a * np.exp(c * x)
